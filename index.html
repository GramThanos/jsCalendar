Load jsCalendar on your project!
<html lang="en">
	<head>
		<meta charset="utf-8">
		<title>jsCalendar</title>
		<!-- jsCalendar Javascript and CSS -->
		<script type="text/javascript" src="jsCalendar.min.js"></script>
		<link rel="stylesheet" type="text/css" href="jsCalendar.min.css">
	</head>
	<body>
		<!-- Calendar Element -->
		<div class="auto-jsCalendar"></div>
	</body>
</html>
